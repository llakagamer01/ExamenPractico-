#include <iostream>
#include <sstream>
#include <string>
#include <cstdlib>
#include <cmath>

using namespace std;

// Headers
string toString (double);
int toInt (string);
double toDouble (string);

int main() {
    cout << "¿Cual es tu Calificacion?" << endl;
    int calif;
    
    cin >> calif;
    if (calif == 100) {
        cout << "A" << endl;
    } else {
        if (calif == 80) {
            cout << "C" << endl;
        } else {
            if (calif == 70) {
                cout << "D" << endl;
            } else {
                cout << "F" << endl;
            }
        }
    }
    return 0;
}

// The following implements type conversion functions.
string toString (double value) { //int also
    stringstream temp;
    temp << value;
    return temp.str();
}

int toInt (string text) {
    return atoi(text.c_str());
}

double toDouble (string text) {
    return atof(text.c_str());
}
