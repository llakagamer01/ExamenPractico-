#include <iostream>
#include <sstream>
#include <string>
#include <cstdlib>
#include <cmath>

using namespace std;

// Headers
string toString (double);
int toInt (string);
double toDouble (string);

int main() {
    cout << "Ingresa tu edad" << endl;
    int edad;
    
    cin >> edad;
    if (edad >= 18) {
        cout << "Eres mayor de edad, puedes pasar" << endl;
    } else {
        cout << "No es mayor de edad, no puedes pasar" << endl;
    }

}

// The following implements type conversion functions.
string toString (double value) { //int also
    stringstream temp;
    temp << value;
    return temp.str();
}

int toInt (string text) {
    return atoi(text.c_str());
}

double toDouble (string text) {
    return atof(text.c_str());
}
